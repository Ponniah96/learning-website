function maxElementArray(){
    const givenArray = [1, 2, 3, 4, 1, 1, 2, 3, 4 ,4 ,4 ,4 ,4];
    var length=0;
    const maxValue = givenArray.reduce((previous, current, i, arr) => {
    if (
        arr.filter(item => item === previous).length >
        arr.filter(item => item === current).length
    ) {
        length=arr.filter(item => item === previous).length
        return previous ;
    } else {
        length=arr.filter(item => item === current).length
        return current;
    }
    });
    //document.getElementById("maxElement").innerText=maxValue;
    //document.getElementById('length').innerText=length;
    console.log(maxValue, length);
}

Array.prototype.upperCase=function(){
    for(let i=0;i<this.length;i++){
        this[i]=this[i].toUpperCase();
    }
}
